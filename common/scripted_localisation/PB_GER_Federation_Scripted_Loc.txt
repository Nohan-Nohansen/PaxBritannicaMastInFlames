defined_text = { 
    name = GER_unrest_impact_rhineland
    
    text = { 
        trigger = {
            OR = {
            check_variable = { Germany_Rhineland_Unrest > 50 }
            check_variable = { Germany_Rhineland_Unrest = 50 }
            }
        }
        localization_key = "GER_unrest_impact_rhineland_high"
    }
    
    text = { 
        trigger = {
            OR = {
            check_variable = { Germany_Rhineland_Unrest > 25 }
            check_variable = { Germany_Rhineland_Unrest = 25 }
            }
        }
        localization_key = "GER_unrest_impact_rhineland_medium"
    }
    
    text = { 
        trigger = {
            OR = {
            check_variable = { Germany_Rhineland_Unrest < 25 }
            check_variable = { Germany_Rhineland_Unrest = 0 }
            }
        }
        localization_key = "GER_unrest_impact_rhineland_low"
    }
}
