PRS_border_Germany = {
	allowed = {
		original_tag = PRS
	}

	enable = {
		GER = {
			exists = yes
		}
	}

	abort_when_not_enabled = yes

	ai_strategy = {
		type = front_unit_request
		tag = GER
		value = 2000
	}
}

dont_border_Poland = {
	allowed = {
		original_tag = PRS
	}
	enable = {
		country_exists = POL
		PRS = {
			NOT = {
				has_war_with = POL
			}
		}
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = ignore
		id = "POL"
		value = 4000
	}

	ai_strategy = {
		type = front_unit_request
		tag = POL
		value = -4000
	}
}
